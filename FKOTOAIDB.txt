-- =========================================================
-- GROUP 1: USER MANAGEMENT
-- =========================================================

CREATE TABLE admins (
  admin_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100) NOT NULL,
  username VARCHAR(50) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  email VARCHAR(150) NOT NULL UNIQUE,
  status ENUM('ACTIVE','INACTIVE') DEFAULT 'ACTIVE',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE students (
  student_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  fullname VARCHAR(100) NOT NULL,
  username VARCHAR(50) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  email VARCHAR(150) NOT NULL UNIQUE,
  avatar_url VARCHAR(500) NULL,
  current_level ENUM('N5','N4') NOT NULL,
  quiz_subscription_expiry DATETIME NULL,
  streak_count INT DEFAULT 0,
  last_login_date DATE NULL,
  rank_points INT DEFAULT 1000,
  status ENUM('ACTIVE','INACTIVE','LOCKED_PENDING') DEFAULT 'ACTIVE',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  INDEX idx_students_status (status),
  INDEX idx_students_rank_points (rank_points),
  INDEX idx_students_level (current_level)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- =========================================================
-- GROUP 2: VOCABULARY & KANJI BANK
-- =========================================================

CREATE TABLE vocabulary (
  vocab_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  word VARCHAR(100) NOT NULL,
  reading VARCHAR(100) NOT NULL,
  meaning TEXT NOT NULL,
  part_of_speech VARCHAR(50) NULL,
  level ENUM('N5','N4') NOT NULL,
  audio_url VARCHAR(500) NULL,
  example_sentence TEXT NULL,
  example_meaning TEXT NULL,
  onyomi VARCHAR(100) NULL,
  kunyomi VARCHAR(100) NULL,
  stroke_order_url VARCHAR(500) NULL,
  is_kanji BOOLEAN DEFAULT FALSE,
  status ENUM('ACTIVE','INACTIVE') DEFAULT 'ACTIVE',
  created_by BIGINT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_vocabulary_created_by
    FOREIGN KEY (created_by) REFERENCES admins(admin_id)
    ON DELETE SET NULL
    ON UPDATE CASCADE,

  INDEX idx_vocab_level (level),
  INDEX idx_vocab_word (word),
  INDEX idx_vocab_is_kanji (is_kanji)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE vocabulary_chapters (
  chapter_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  chapter_name VARCHAR(100) NOT NULL,
  level ENUM('N5','N4') NOT NULL,
  order_index INT NOT NULL,
  description TEXT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE vocabulary_chapter_items (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  chapter_id BIGINT NOT NULL,
  vocab_id BIGINT NOT NULL,
  order_index INT NOT NULL,

  CONSTRAINT fk_vci_chapter
    FOREIGN KEY (chapter_id) REFERENCES vocabulary_chapters(chapter_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

  CONSTRAINT fk_vci_vocab
    FOREIGN KEY (vocab_id) REFERENCES vocabulary(vocab_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

  UNIQUE KEY uq_vci_chapter_vocab (chapter_id, vocab_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE user_word_metrics (
  metric_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  student_id BIGINT NOT NULL,
  vocab_id BIGINT NOT NULL,
  mastered BOOLEAN DEFAULT FALSE,
  flashcard_correct INT DEFAULT 0,
  flashcard_incorrect INT DEFAULT 0,
  quiz_correct INT DEFAULT 0,
  quiz_incorrect INT DEFAULT 0,
  total_wrong_count INT GENERATED ALWAYS AS (quiz_incorrect + flashcard_incorrect) STORED,
  last_reviewed_at DATETIME NULL,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_uwm_student
    FOREIGN KEY (student_id) REFERENCES students(student_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

  CONSTRAINT fk_uwm_vocab
    FOREIGN KEY (vocab_id) REFERENCES vocabulary(vocab_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

  UNIQUE KEY uq_uwm_student_vocab (student_id, vocab_id),
  INDEX idx_uwm_student_wrong (student_id, total_wrong_count DESC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- =========================================================
-- GROUP 3: ARTICLES / READING COMPREHENSION
-- =========================================================

CREATE TABLE articles (
  article_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  title VARCHAR(255) NOT NULL,
  content LONGTEXT NOT NULL,
  content_tokenized JSON NULL,
  level ENUM('N5','N4') NOT NULL,
  source VARCHAR(255) NULL,
  thumbnail_url VARCHAR(500) NULL,
  status ENUM('ACTIVE','INACTIVE') DEFAULT 'ACTIVE',
  created_by BIGINT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_articles_created_by
    FOREIGN KEY (created_by) REFERENCES admins(admin_id)
    ON DELETE SET NULL
    ON UPDATE CASCADE,

  INDEX idx_articles_level (level),
  INDEX idx_articles_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- =========================================================
-- GROUP 4: EXAM & QUIZ SYSTEM
-- =========================================================

CREATE TABLE exam_categories (
  category_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  category_name VARCHAR(100) NOT NULL,
  category_type ENUM('MOCK_EXAM','TOPIC_QUIZ') NOT NULL,
  level ENUM('N5','N4') NOT NULL,
  parent_id BIGINT NULL,
  order_index INT DEFAULT 0,
  status ENUM('ACTIVE','INACTIVE') DEFAULT 'ACTIVE',

  CONSTRAINT fk_exam_categories_parent
    FOREIGN KEY (parent_id) REFERENCES exam_categories(category_id)
    ON DELETE SET NULL
    ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE questions (
  question_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  category_id BIGINT NOT NULL,
  question_text TEXT NOT NULL,
  question_image_url VARCHAR(500) NULL,
  audio_url VARCHAR(500) NULL,
  option_a TEXT NOT NULL,
  option_b TEXT NOT NULL,
  option_c TEXT NOT NULL,
  option_d TEXT NOT NULL,
  correct_answer ENUM('A','B','C','D') NOT NULL,
  explanation TEXT NOT NULL,
  level ENUM('N5','N4') NOT NULL,
  related_vocab_id BIGINT NULL,
  status ENUM('ACTIVE','INACTIVE') DEFAULT 'ACTIVE',
  created_by BIGINT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_questions_category
    FOREIGN KEY (category_id) REFERENCES exam_categories(category_id)
    ON DELETE RESTRICT
    ON UPDATE CASCADE,

  CONSTRAINT fk_questions_related_vocab
    FOREIGN KEY (related_vocab_id) REFERENCES vocabulary(vocab_id)
    ON DELETE SET NULL
    ON UPDATE CASCADE,

  CONSTRAINT fk_questions_created_by
    FOREIGN KEY (created_by) REFERENCES admins(admin_id)
    ON DELETE SET NULL
    ON UPDATE CASCADE,

  INDEX idx_questions_category (category_id),
  INDEX idx_questions_level (level)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
